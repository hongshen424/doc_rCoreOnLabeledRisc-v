2.火苗平台简介

火苗平台简介

火苗系统是中国科学院计算技术研究所研发的，基于思沃（SERVE）平台下的云服务版（SEVER.c）以及标签化冯诺伊曼体系结构版（SERVE.v）的两者结合的FPGA平台。具有低成本，价格低于¥1000；低功耗，USB接口直接供电；低门槛，开源软硬件设计；高可靠，直接使用ARM SoC外设控制器IP硬核；在线使用，通过标准以太网ssh远程登录使用等特点。SERVE.v版本则是Labeled RISC-V FPGA的原型系统配置版本。拥有4个rocket核心。16KB的一级指令缓存，16KB 的一级数据缓存，2MB的二级缓存。千兆以太网。可运行Redis、Memcached等应用。当然，支持标签化特色的性能调控技术。

多核:操作系统3种处理模式(SMP+AMP+BMP)

cat ~/master/nohype_run`

#!/usr/bin/env bash

sysctl net.ipv4.ip_forward=0
./nohype_settings 5
./4-core-reset linux.nohype 0.dtb 1.dtb 2.dtb 3.dtb 0x00000000 0x20000000 0x40000000 0x60000000

分别解释这两条命令

./nohype_settings 和./4-core-reset

1、在不加参数执行./nohype_settings 时

distinct dsid [*], reset to hang [ ], mem partition [*]

表示的含义是：在nohype模式之下，复位的效果是

distinct dsid [*]:每个核使用不同的dsid，

reset to hang [ ]:设置的两种复位方案，一种是正常复位，一种是复位到bootroom里面一个死循环，

`mem partition : 把内存参数设置成分段

5的含义就是二进制下的101，分别对应三个中括号，1是ture，0是false。

2、./4-core-reset

让我们来看一下源码

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <memory.h>
#include <unistd.h>  
#include <sys/mman.h>  
#include <sys/types.h>  
#include <sys/stat.h>  
#include <fcntl.h>
#include <elf.h>

#include <assert.h>
#include <sys/time.h>
#include <time.h>

#define DDR_TOTAL_SIZE        0x80000000
#define DDR_BASE_ADDR        ((uintptr_t)0x800000000)

#define GPIO_RESET_TOTAL_SIZE    0x1000
#define GPIO_RESET_BASE_ADDR    0x80010000

void *ddr_base;
volatile uint32_t *gpio_reset_base;
int    fd;

void loader(char *imgfile, char *dtbfile, int offset) {
    FILE *fp = fopen(imgfile, "rb");
    assert(fp);

    fseek(fp, 0, SEEK_END);
    long size = ftell(fp);
    printf("image size = %ld\n", size);

    fseek(fp, 0, SEEK_SET);
    size_t nr_read = fread(ddr_base + offset, size, 1, fp);
    printf("payload size %zd\n", nr_read);

    fclose(fp);

    fp = fopen(dtbfile, "rb");
    if (fp == NULL) {
        printf("No valid dtb file provided. Dtb in bootrom will be used.\n");
        return ;
    }

    fseek(fp, 0, SEEK_END);
    size = ftell(fp);
    printf("dtb size = %ld\n", size);

    fseek(fp, 0, SEEK_SET);
    fread(ddr_base + offset + 0x8, size, 1, fp);

    fclose(fp);
}

void* create_map(size_t size, int fd, off_t offset) {
    void *base = mmap(NULL, size, PROT_READ|PROT_WRITE, MAP_SHARED, fd, offset);

    if (base == NULL) {
        perror("init_mem mmap failed:");
        close(fd);
        exit(1);
    }

    return base;
}

void init_map() {
    fd = open("/dev/mem", O_RDWR|O_SYNC);  
    if (fd == -1)  {  
        perror("init_map open failed:");
        exit(1);
    } 

    gpio_reset_base = create_map(GPIO_RESET_TOTAL_SIZE, fd, GPIO_RESET_BASE_ADDR);
    ddr_base = create_map(DDR_TOTAL_SIZE, fd, DDR_BASE_ADDR);
}

void resetn(int val) {
    gpio_reset_base[0] = val;
}

void finish_map() {
    munmap((void *)gpio_reset_base, GPIO_RESET_TOTAL_SIZE);
    munmap((void *)ddr_base, DDR_TOTAL_SIZE);
    close(fd);
}

int main(int argc, char *argv[]) {
    /* map some devices into the address space of this program */
    init_map();

    /* reset RISC-V cores */
    resetn(0);

    printf("%s %s %s\n", argv[1], argv[2], argv[6]);
    loader(argv[1], argv[2], strtoll(argv[6], NULL, 16));
    printf("%s %s %s\n", argv[1], argv[3], argv[7]);
    loader(argv[1], argv[3], strtoll(argv[7], NULL, 16));
    printf("%s %s %s\n", argv[1], argv[4], argv[8]);
    loader(argv[1], argv[4], strtoll(argv[8], NULL, 16));
    printf("%s %s %s\n", argv[1], argv[5], argv[9]);
    loader(argv[1], argv[5], strtoll(argv[9], NULL, 16));

    /* finish resetting RISC-V cores */
    resetn(3);

    finish_map();

    return 0; 
}

0、启动os的流程

labeled risc-v 提供了一个只有hello world最小化的linux，我们从这个入手来分析启动的流程。

Build RISC-V image

  • Build riscv-pk (which contains the bootloader), riscv-linux and riscv-rootfs with the following commands
mkdir build
make -j sw  # change 16 to the number of cores according to your host

If it is the first time you run this command, note that you will be asked to pull these repos under ../../sw.

After that, linux.bin will be generated under build/. This is a minimal rootfs only containing a hello program.

经过这一步会在build文件夹下,生成linux.bin，此linux.bin可以在火苗上直接运行，怎么运行呢？

首先利用前面提到过的通过scp命令来进行本地和火苗之间的通信，

$ scp -P 2234 linux.bin thu_chenyu@159.226.41.100:~
$ scp linux.bin fidus84:/root/master/

$ scp -P 2234 kernel.bin thu_chenyu@159.226.41.100:~
$ scp kernel.bin fidus84:/root/master/

这个时候在/master目录下就有了linux.bin 文件。此文件就是可以跑的操作系统。

那么怎么运行起来呢？

1、写一个脚本

仿照smp_run写一个脚本。

#!/usr/bin/env bash

#sysctl net.ipv4.ip_forward=0
./nohype_settings 0
./4-core-reset linux.bin 4.dtb.small 1.dtb 2.dtb 3.dtb 0x00000000 0x20000000 0x40000000 0x60000000

#sh nat.sh
#ssh-keygen -f "/root/.ssh/known-hosts" -R 192.168.1.2

2、开启串口

回到/root路径下，开启串口

$ sh minicom.sh

3、运行脚本

注意！：另打开一个终端连上火苗，运行刚刚写的脚本

$ cd master
$ sh kernel_run

说明linux.bin 已经跑起来了。
